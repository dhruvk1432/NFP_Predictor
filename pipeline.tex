\section{Scope}
This document captures the production path to keep:
\begin{enumerate}
    \item Build SA revised and NSA revised LightGBM branches.
    \item Build NSA + predicted seasonal adjustment.
    \item Build the dynamic SA blend champion.
    \item Fuse blend champion with consensus anchor using A7 Kalman fusion.
\end{enumerate}
Only Kalman fusion is retained as the consensus-anchor model in this document.

\section{Notation and Data Objects}
\subsection{Notation}
\begin{align*}
 y_t^{SA} &: \text{true SA revised MoM target at month } t,\\
 y_t^{NSA} &: \text{true NSA revised MoM target at month } t,\\
 \hat{y}_{t,SA} &: \text{direct SA revised LightGBM prediction},\\
 \hat{y}_{t,NSA} &: \text{direct NSA revised LightGBM prediction},\\
 a_t &: y_t^{SA}-y_t^{NSA},\\
 \hat{a}_t &: \text{predicted seasonal adjustment},\\
 \hat{y}_{t,adj} &: \hat{y}_{t,NSA}+\hat{a}_t,\\
 \hat{y}_{t,blend} &: \text{walk-forward blend of }\hat{y}_{t,SA}\text{ and }\hat{y}_{t,adj},\\
 c_t &: \text{consensus anchor value},\\
 m_t &: \text{model observation in fusion (here }\hat{y}_{t,blend}\text{)}.
\end{align*}

\subsection{Point-in-time safety policy}
For every target month $t$:
\begin{itemize}
    \item Training data uses only observations available before the prediction cutoff for $t$.
    \item Revised targets enforce operational availability timing.
    \item Backtesting uses strict expanding-window chronology (no lookahead).
\end{itemize}

\section{LightGBM Branches: Detailed Configuration}
\subsection{Target branches retained in this path}
\begin{itemize}
    \item \texttt{sa\_first\_revised}: SA revised target branch.
    \item \texttt{nsa\_first\_revised}: NSA revised target branch.
\end{itemize}
Both are trained using revised-target caches from \texttt{data/NFP\_target}.

\subsection{Feature construction pipeline}
For each target month, the feature vector is built from:
\begin{itemize}
    \item Feature-selected master snapshots (pre-merged macro/exogenous sources).
    \item Branch-target lag features (lag-1/2/3/6/12), rolling moments, volatility, acceleration/jerk transforms.
    \item Revision features from cross-snapshot differences (master[M] vs master[M-1]).
    \item Calendar/timing features (for SA branch, seasonal encodings are filtered to structural subset).
\end{itemize}

\subsection{Core LightGBM training configuration}
Base LightGBM defaults:
\begin{itemize}
    \item Objective: regression (Huber disabled by default in current config).
    \item Learning rate: 0.03.
    \item Num leaves: 31.
    \item Max depth: 6.
    \item Min data in leaf: 5.
    \item Feature fraction: 0.8.
    \item Bagging fraction/freq: 0.8 / 5.
\end{itemize}
Training controls:
\begin{itemize}
    \item Time-series CV splits: 5.
    \item Max boosting rounds: 1000.
    \item Early stopping rounds: 50.
\end{itemize}

\subsection{Recency weighting and optional tail weighting}
Each training sample is weighted by recency:
\begin{align}
w_i \propto \exp\left(-\ln(2)\frac{\Delta_i}{h}\right),
\end{align}
where $\Delta_i$ is month distance to target and $h$ is half-life.
Weights are normalized to mean 1.0.

Tail-aware multipliers can be applied (for variance-priority targets) based on large level and large change quantiles, with capped maximum multiplier.

\subsection{Feature selection stack at train time}
The branch uses a layered selection policy:
\begin{itemize}
    \item Snapshot base feature set from selected-feature caches.
    \item Optional union-pool short-pass (top-K, weighted ranker).
    \item Branch-target feature selector (method varies by branch priority).
    \item Always-keep blocks (revision features and structural calendar fields).
\end{itemize}

\subsection{Tuning and gating}
\begin{itemize}
    \item Hyperparameters are tuned with Optuna in a time-series CV setup.
    \item Keep-rule checks trailing OOS MAE vs baselines.
    \item Variance gate can block production eligibility if variance-capture KPIs are below thresholds.
\end{itemize}

\subsection{Backtest protocol}
For each backtest month $t$:
\begin{enumerate}
    \item Train on all prior valid rows.
    \item Recompute feature set for that window.
    \item Tune (when scheduled), then fit model.
    \item Predict month $t$ OOS.
\end{enumerate}

\section{NSA + Predicted Seasonal Adjustment Layer}
\subsection{Definition}
\begin{align}
a_t &= y_t^{SA}-y_t^{NSA}, \\
\hat{y}_{t,adj} &= \hat{y}_{t,NSA}+\hat{a}_t.
\end{align}

\subsection{Model family and selected predictor}
Candidate adjustment predictors:
\begin{enumerate}
    \item SARIMA,
    \item Monthly average,
    \item 12-month complement,
    \item Same-month last year,
    \item Exponentially weighted same-month average,
    \item Month-dummy linear regression with trend.
\end{enumerate}
Current selected model is \texttt{exp\_weighted\_avg}.

\subsection{Selected adjustment formula}
Using same-calendar-month historical rows $j$ in PIT-safe history:
\begin{align}
u_j &= \exp\left(-\ln(2)\frac{\text{years\_ago}_j}{3.0}\right),\\
\omega_j &= \frac{u_j}{\sum_k u_k},\\
\hat{a}_t &= \sum_j \omega_j a_j.
\end{align}

\section{SA Blend Champion Layer}
\subsection{Blend equation}
\begin{align}
\hat{y}_{t,blend}=w_t\hat{y}_{t,SA}+(1-w_t)\hat{y}_{t,adj}.
\end{align}

\subsection{Walk-forward weight search}
\begin{itemize}
    \item If available history $<12$, set $w_t=0.5$.
    \item Otherwise search $w_t\in\{0,0.05,\dots,1\}$ over trailing 18-month history.
\end{itemize}
Objective:
\begin{align}
\mathcal{L} =
\text{MAE}
&+18\lvert1-\text{std\_ratio}\rvert
+18\lvert1-\text{diff\_std\_ratio}\rvert \nonumber\\
&+0.20\cdot\text{tail\_mae}
+14\cdot(1-\text{corr\_diff})
+10\cdot(1-\text{diff\_sign\_accuracy}).
\end{align}

\section{A7 Kalman Fusion (Retained Consensus-Anchor Model)}
\subsection{State-space model}
\begin{align}
x_t &= x_{t-1}+w_t, & w_t &\sim\mathcal{N}(0,Q_t),\\
c_t &= x_t+v_{c,t}, & v_{c,t} &\sim\mathcal{N}(0,R_{c,t}),\\
m_t &= x_t+v_{m,t}, & v_{m,t} &\sim\mathcal{N}(0,R_{m,t}).
\end{align}

\subsection{Information-form filter update}
\begin{align}
P_{t|t-1} &= P_{t-1}+Q_t,\\
I_t &= \frac{1}{P_{t|t-1}}+\frac{1}{R_{c,t}}+\frac{1}{R_{m,t}},\\
P_{t|t} &= \frac{1}{I_t},\\
\hat{y}_t = x_{t|t} &= P_{t|t}\left(\frac{x_{t|t-1}}{P_{t|t-1}}+\frac{c_t}{R_{c,t}}+\frac{m_t}{R_{m,t}}\right).
\end{align}

\subsection{Noise adaptation rules}
\begin{itemize}
    \item Warm start from full consensus history:
    \begin{align*}
    R_{c,\text{init}} &= \operatorname{Var}(y-c)\text{ over last 60 months},\\
    Q_{\text{init}} &= \operatorname{Var}(\Delta y)\text{ over last 60 months}.
    \end{align*}
    \item If at least 6 historical overlap points exist, update $R_c,R_m,Q$ from trailing windows (up to 18 rows).
\end{itemize}

Initialization values from current artifacts:
\begin{align*}
\text{First-release anchor:}\quad &R_{c,\text{init}}=30534.3669,\quad Q_{\text{init}}=41691.3472,\\
\text{Latest anchor:}\quad &R_{c,\text{init}}=29897.4743,\quad Q_{\text{init}}=41691.3472.
\end{align*}

\subsection{Teacher-forced state update on realized months}
After each realized month:
\begin{align}
x_t \leftarrow y_t,\qquad P_t \leftarrow 10^{-6}.
\end{align}
