\section{Purpose}
This document describes the end-to-end forecasting pipeline currently used in this repository:
\begin{enumerate}
    \item Build direct SA revised and NSA revised prediction streams.
    \item Build NSA + predicted seasonal adjustment stream.
    \item Build dynamic SA blend champion.
    \item Use consensus as an anchor and run fusion models (including A5 and A7).
\end{enumerate}
It includes the core mathematics, data construction rules, and metrics-based analysis.

\section{Data, Targets, and Notation}
\subsection{Notation}
\begin{align*}
y_t^{SA} &: \text{true SA revised MoM target at month } t \\
y_t^{NSA} &: \text{true NSA revised MoM target at month } t \\
\hat{y}_{t,\text{SA}} &: \text{direct SA revised model prediction} \\
\hat{y}_{t,\text{NSA}} &: \text{direct NSA revised model prediction} \\
a_t &: \text{seasonal adjustment term } (y_t^{SA} - y_t^{NSA}) \\
\hat{a}_t &: \text{predicted seasonal adjustment term} \\
\hat{y}_{t,\text{adj}} &: \hat{y}_{t,\text{NSA}} + \hat{a}_t \\
\hat{y}_{t,\text{blend}} &: \text{dynamic blend of } \hat{y}_{t,\text{SA}} \text{ and } \hat{y}_{t,\text{adj}} \\
c_t &: \text{consensus anchor value} \\
m_t &: \text{model observation in consensus fusion (here } \hat{y}_{t,\text{blend}} \text{)}.
\end{align*}

\subsection{Point-in-time safety}
All model stages use expanding-window logic and strict chronology:
\begin{itemize}
    \item Training rows for month $t$ only use information available before or at the configured point-in-time cutoff for $t$.
    \item For revised targets and adjustment histories, operational availability dates are enforced.
\end{itemize}

\section{Stage 1: Direct SA Revised and NSA Revised Models}
\subsection{Targets}
Two revised target branches are used:
\begin{itemize}
    \item SA revised: \texttt{sa\_first\_revised}
    \item NSA revised: \texttt{nsa\_first\_revised}
\end{itemize}
Targets are loaded from pre-built revised parquet caches under \texttt{data/NFP\_target}.

\subsection{Feature construction}
For each target month, the feature vector includes:
\begin{itemize}
    \item Master snapshot features (pre-selected feature sets).
    \item Branch-target lag/rolling/acceleration features.
    \item Calendar/timing features (SA branch keeps structural subset only).
    \item Revision features comparing current and previous snapshot views.
\end{itemize}

\subsection{Training protocol}
Backtesting is expanding-window and monthly:
\begin{itemize}
    \item Train on all valid historical rows before month $t$.
    \item Refit model each step.
    \item Hyperparameters tuned (periodically) with time-series CV.
    \item Sample weighting uses exponential recency decay:
\end{itemize}
\begin{align}
w_i \propto \exp\left(-\ln(2)\frac{\Delta_i}{h}\right),
\end{align}
where $\Delta_i$ is distance in months from sample $i$ to target month and $h$ is half-life.

\section{Stage 2: NSA + Predicted Seasonal Adjustment}
\subsection{Construction}
Define adjustment:
\begin{align}
a_t = y_t^{SA} - y_t^{NSA}.
\end{align}
Then define adjusted SA-space prediction:
\begin{align}
\hat{y}_{t,\text{adj}} = \hat{y}_{t,\text{NSA}} + \hat{a}_t.
\end{align}

\subsection{Adjustment model family}
Candidate models tested:
\begin{enumerate}
    \item SARIMA
    \item Monthly average
    \item 12-month complement
    \item Same-month last year
    \item Exponentially weighted same-month average
    \item Linear regression with month dummies and trend
\end{enumerate}

Current selected model is \texttt{exp\_weighted\_avg} (by SA composite score).

\subsection{Selected adjustment formula}
For target month $t$, using same-calendar-month historical rows $j$ in PIT-safe history:
\begin{align}
u_j &= \exp\left(-\ln(2)\frac{\text{years\_ago}_j}{3.0}\right), \\
\omega_j &= \frac{u_j}{\sum_k u_k}, \\
\hat{a}_t &= \sum_j \omega_j a_j.
\end{align}

\section{Stage 3: SA Blend Champion}
\subsection{Blend equation}
\begin{align}
\hat{y}_{t,\text{blend}} = w_t \hat{y}_{t,\text{SA}} + (1-w_t)\hat{y}_{t,\text{adj}}.
\end{align}

\subsection{Walk-forward weight selection}
\begin{itemize}
    \item If history $<$ \texttt{min\_history} (12), set $w_t=0.5$.
    \item Else choose $w_t \in \{0,0.05,\dots,1\}$ minimizing objective over trailing window (18 months).
\end{itemize}

Composite objective:
\begin{align}
\mathcal{L} =
\text{MAE}
&+ 18\lvert 1-\text{std\_ratio}\rvert
+ 18\lvert 1-\text{diff\_std\_ratio}\rvert \nonumber\\
&+ 0.20\cdot \text{tail\_mae}
+ 14\cdot(1-\text{corr\_diff})
+ 10\cdot(1-\text{diff\_sign\_accuracy}).
\end{align}

\section{Consensus-Anchor Models of Interest}
\subsection{A5: Acceleration Override}
Given last known actual $y_{t-1}$:
\begin{align}
\Delta_m &= m_t - y_{t-1}, \\
\Delta_c &= c_t - y_{t-1}.
\end{align}
If $\operatorname{sign}(\Delta_m)=\operatorname{sign}(\Delta_c)$, prediction reverts to consensus:
\begin{align}
\hat{y}_t = c_t.
\end{align}
If signs disagree:
\begin{itemize}
    \item \textbf{consensus mode}: $\hat{y}_t = y_{t-1} + \operatorname{sign}(\Delta_m)|\Delta_c|$
    \item \textbf{blend mode}: $\hat{y}_t = y_{t-1} + \operatorname{sign}(\Delta_m)\left(\kappa|\Delta_m| + (1-\kappa)|\Delta_c|\right)$
\end{itemize}
This document focuses on:
\begin{itemize}
    \item \texttt{A5\_AccelOverride\_consensus\_k0.70}
    \item \texttt{A5\_AccelOverride\_blend\_k0.30}
\end{itemize}

\subsection{A7: Kalman Fusion}
State-space specification:
\begin{align}
x_t &= x_{t-1} + w_t, \quad w_t \sim \mathcal{N}(0,Q_t), \\
c_t &= x_t + v_{c,t}, \quad v_{c,t} \sim \mathcal{N}(0,R_{c,t}), \\
m_t &= x_t + v_{m,t}, \quad v_{m,t} \sim \mathcal{N}(0,R_{m,t}).
\end{align}

Information-form update:
\begin{align}
P_{t|t-1} &= P_{t-1} + Q_t, \\
I_t &= \frac{1}{P_{t|t-1}} + \frac{1}{R_{c,t}} + \frac{1}{R_{m,t}}, \\
P_{t|t} &= \frac{1}{I_t}, \\
\hat{y}_t = x_{t|t} &= P_{t|t}\left(
\frac{x_{t|t-1}}{P_{t|t-1}} + \frac{c_t}{R_{c,t}} + \frac{m_t}{R_{m,t}}
\right).
\end{align}

Initialization (first-release anchor run):
\begin{align*}
R_{c,\text{init}} &= 30534.3669, \\
Q_{\text{init}} &= 41691.3472.
\end{align*}
Initialization (latest-consensus anchor run):
\begin{align*}
R_{c,\text{init}} &= 29897.4743, \\
Q_{\text{init}} &= 41691.3472.
\end{align*}

After each realized month, state is teacher-forced to observed truth:
\begin{align}
x_t \leftarrow y_t, \quad P_t \leftarrow 10^{-6}.
\end{align}

\section{Interpretation}
\begin{itemize}
    \item A5 variants often tie consensus exactly by construction (when model and consensus directional signal agrees, A5 outputs consensus directly).
    \item A7 uses both anchor and blend signals continuously, with adaptive confidence weighting; this tends to produce materially higher strict win rates against consensus.
    \item Under both anchor definitions, A7 is the strongest of the three on trailing MAE windows.
\end{itemize}
